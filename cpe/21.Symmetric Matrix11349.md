#include<iostream>
using namespace std;
int main()
{
	long long t, n;
	char q, y;
	cin >>t;
	int w = 1;
	while (t--)
	{
		bool r = false;
		long long a[300][300];
		
		cin >> q>>y>>n;
		for (int i = 0; i < n; i++)
		{
			for (int j = 0; j < n; j++)
			{
				cin>>a[i][j];
				if (a[i][j] < 0)
				{
					r = true;
				}
			}
		}
		for (int k = 0; k < n; k++)
		{
			if (!r)
			{
				for (int l = 0; l < n ; l++)
				{
					if (a[k][l] != a[n-1 - k][n-1- l])
					{
						r = true;
						break;
					}

						
				}
			}
		}
			if (r)
			{
				cout << "Test #" << w << ": Non-symmetric." << endl;
			}
			else
			{
				cout << "Test #" << w << ": Symmetric." << endl;
			}
			w++;
	}
}
