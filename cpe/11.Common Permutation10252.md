```c
#include<iostream>
#include<algorithm>

using namespace std;
int main()
{
	string a, b;
	while (cin >> a >> b)
	{
		int aa[26] = { 0 };
		int bb[26] = { 0 };
		for (int i = 0; i < a.size(); i++)
		{
			aa[a[i]-'a']++;
		}
		for (int j = 0; j < b.size(); j++)
		{
			bb[b[j] - 'a']++;
		}
		int o;
		for (int k = 0; k < 26; k++)
		{
		    int o = min(aa[k], bb[k]);
			for (int l = 0; l < o; l++)
			{
				cout << char(k+'a');
			}
	    }
		cout << endl;
	}
}
```
