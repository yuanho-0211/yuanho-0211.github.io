#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int a, b,c,d;
	while (cin >> a >> b)
	{
		if (a == 0 && b == 0)
		{
			break;
		}
		c = sqrt(a);
		d = 0;
		if (a == c*c)
		{
			d++;
		}
		for (int i = a; i <= b; i++)
		{
			
			if (sqrt(i) == c + 1)
			{
				d++;
				c = sqrt(i);

			}
		}
		cout << d << endl;
	}
}
