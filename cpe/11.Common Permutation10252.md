```c
#include<iostream>
#include<algorithm>

using namespace std;
int main()
{
	string a, b;
	while (cin >> a >> b)
	{
		int aa[26] = { 0 };
		int bb[26] = { 0 };
		for (int i = 0; i < a.size(); i++)
		{
			aa[a[i]-'a']++;
		}
		for (int j = 0; j < b.size(); j++)
		{
			bb[b[j] - 'a']++;
		}
		int o;
		for (int k = 0; k < 26; k++)
		{
		    int o = min(aa[k], bb[k]);
			for (int l = 0; l < o; l++)
			{
				cout << char(k+'a');
			}
	    }
		cout << endl;
	}
}
```
- 1 要考慮到重複的
## 2 用兩個陣列分別計算字母數量
## 3 數量取小的輸出，用for寫就可以控制每個字母要輸出幾個，也會自動不輸出0的
