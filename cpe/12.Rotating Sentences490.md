```c
#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
#include<cctype>
using namespace std;
int main()
{
	string s;
	int o;
	int n = 0;
	int a,b;
	vector <string> c;
	o = 0;
	while (getline(cin, s))
	{
		
		c.push_back(s);
		 a = c[o].size();
		 b = c[o].size();
		if (b > n)
		{
			n = b;
		}
		o++;
		
	}
	for (int k = 0; k <n; k++)
	{
		for (int i = o - 1; i >=0; i--)
		{
			if (k<c[i].size())
			{
				cout << c[i][k];
			}
			else
			{
				cout << ' ';
			}
			
		}
		cout << endl;
	}
}
```
