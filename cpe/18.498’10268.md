#include <iostream>
#include <vector>
#include <cmath>
#include <string>
#include <sstream>
using namespace std;
int main()
{
	long long x,a;
	while (cin >> x)
	{
		vector<long long>v;
		long long sum = 0;
		string line;
			cin.ignore();
			getline(cin, line);

			stringstream ss(line);
			while (ss >> a)
			{
				v.push_back(a);
			}

		long long b = v.size()-1;
		for (int i = 0; i <b; i++)
		{
			sum = sum + v[i] * (b - i) * (pow(x, b -i-1));
		}
		cout << sum << endl;
	}
}
